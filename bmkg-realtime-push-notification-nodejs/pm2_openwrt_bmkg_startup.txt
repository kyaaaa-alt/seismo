#!/bin/sh /etc/rc.common

START=99

USE_PROCD=1
NAME=bmkg
PROG=/usr/bin/pm2

start_service() {
        procd_open_instance
        procd_set_param command "$PROG" start /root/seismo/index.js --watch
        procd_set_param respawn
        procd_set_param stdout 1
        procd_set_param stderr 1
        procd_close_instance
}
